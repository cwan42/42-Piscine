void	ft_ultimate_div_mod(int *a, int *b)
{
	int	div;
	int	mod;

	div = *a / *b;
	mod = *a % *b;
	*a = div;
	*b = mod;
}

#include <unistd.h>

void  ft_putchar(char c)
{
      write(1, &c, 1);
}

void  ft_putstr(char *str)
{
      int i;

      i = 0;
      while (str[i] != '\n')
      {
            ft_putchar(str[i]);
            i++;
      }
}
